161  public float mad(List track) {
162 
163  int length = track.size();
164  ListIterator jT = track.listIterator();
165  Spermatozoon oldSpermatozoon = (Spermatozoon) jT.next();
166  float totalDegrees = 0;
167  for (int i = 1; i < length; i++) {
168  Spermatozoon newSpermatozoon = (Spermatozoon) track.get(i);
169  float diffX = newSpermatozoon.x - oldSpermatozoon.x;
170  float diffY = newSpermatozoon.y - oldSpermatozoon.y;
171  double angle = (2 * Math.PI + Math.atan2(diffY, diffX)) % (2 * Math.PI);
172  totalDegrees += angle;
173  oldSpermatozoon = newSpermatozoon;
174  }
175  // mean angle
176  float meanAngle = totalDegrees / (length - 1);
177  return meanAngle;
178  }

230  public float vcl(List track) {
231 
232  int length = track.size();
233  ListIterator jT = track.listIterator();
234  Spermatozoon oldSpermatozoon = (Spermatozoon) jT.next();
235  float distance = 0;
236  for (; jT.hasNext();) {
237  Spermatozoon newSpermatozoon = (Spermatozoon) jT.next();
238  distance += newSpermatozoon.distance(oldSpermatozoon);
239  oldSpermatozoon = newSpermatozoon;
240  }
241  // convert pixels to micrometers
242  distance = distance * (float) Params.micronPerPixel;
243  // Seconds
244  float elapsedTime = (length - 1) / Params.frameRate;
245  // return um/second
246  return distance / elapsedTime;
247  }
