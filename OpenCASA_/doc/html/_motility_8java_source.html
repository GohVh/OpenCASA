161  private void calculateAverageMotility(ResultsTable rt, Trial trial) {
162 
163  SignalProcessing sp = new SignalProcessing();
164  List<List<Spermatozoon>> filteredTracks = sp.filterTracksByMotility(trial.tracks);
165  float nTracks = filteredTracks.size(); // Only take into account those who passed the motility test
166  float vsl_mean = total_vsl / nTracks;
167  float vcl_mean = total_vcl / nTracks;
168  float vap_mean = total_vap / nTracks;
169  float lin_mean = total_lin / nTracks;
170  float wob_mean = total_wob / nTracks;
171  float str_mean = total_str / nTracks;
172  float alhMean_mean = total_alhMean / nTracks;
173  float alhMax_mean = total_alhMax / nTracks;
174  float bcf_mean = total_bcf / nTracks;
175  float dance_mean = total_dance / nTracks;
176  float mad_mean = total_mad / nTracks;
177  // % progressive Motile sperm
178  float progressiveMotPercent = countProgressiveSperm / (float) nTracks;
179  // % motility
180  Kinematics K = new Kinematics();
181  int[] motileSperm = K.motilityTest(trial.tracks);
182  int countMotileSperm = motileSperm[0];
183  total_motile += countMotileSperm;
184  int countNonMotileSperm = motileSperm[1];
185  total_nonMotile += countNonMotileSperm;
186  float motility_value = (float) countMotileSperm / ((float) (countMotileSperm + countNonMotileSperm));
187  total_sperm += nTracks;
188 
189  rt.incrementCounter();
190  rt.addValue("Motile trajectories", nTracks);
191  rt.addValue("VSL Mean (um/s)", vsl_mean);
192  rt.addValue("VCL Mean (um/s)", vcl_mean);
193  rt.addValue("VAP Mean (um/s)", vap_mean);
194  rt.addValue("LIN Mean ", lin_mean);
195  rt.addValue("WOB Mean ", wob_mean);
196  rt.addValue("STR Mean ", str_mean);
197  rt.addValue("ALH_Mean Mean (um)", alhMean_mean);
198  rt.addValue("ALH_Max Mean (um)", alhMax_mean);
199  rt.addValue("BCF Mean (Hz)", bcf_mean);
200  rt.addValue("DANCE Mean (um^2/s)", dance_mean);
201  rt.addValue("MAD Mean (degrees)", mad_mean);
202  rt.addValue("Progressive Motility (%)", progressiveMotPercent * 100);
203  rt.addValue("Motility (%)", motility_value * 100);
204  rt.addValue("Sample", trial.type);
205  rt.addValue("ID", trial.ID);
206  rt.addValue("Source", trial.source);
207  if (!Params.male.isEmpty())
208  rt.addValue("Male", Params.male);
209  if (!Params.date.isEmpty())
210  rt.addValue("Date", Params.date);
211  if (!Params.genericField.isEmpty())
212  rt.addValue("Generic Field", Params.genericField);
213  }

223  private void calculateMotility(ResultsTable rt, Trial trial) {
224 
225  SignalProcessing sp = new SignalProcessing();
226  Kinematics K = new Kinematics();
227  // Only pass from here tracks with a minimum level of motility
228  List<List<Spermatozoon>> filteredTracks = sp.filterTracksByMotility(trial.tracks);
229  // Calculate values for each track
230  for (ListIterator iT = filteredTracks.listIterator(); iT.hasNext();) {
231  List aTrack = (List) iT.next();
232  List avgTrack = sp.movingAverage(aTrack);
233  float length = (float) aTrack.size();
234  // VSL
235  float vsl_value = K.vsl(aTrack);
236  total_vsl += vsl_value;
237  // VCL
238  float vcl_value = K.vcl(aTrack);
239  total_vcl += vcl_value;
240  // VAP is equivalent to calculate vcl from averaged track
241  float vap_value = K.vcl(avgTrack);
242  total_vap += vap_value;
243  // Linearity
244  float lin_value = (vsl_value / vcl_value) * 100;
245  total_lin += lin_value;
246  // Wobble
247  float wob_value = (vap_value / vcl_value) * 100;
248  total_wob += wob_value;
249  // Straightness
250  float str_value = (vsl_value / vap_value) * 100;
251  total_str += str_value;
252  // Amplitude of lateral head
253  float[] alh_values = K.alh(aTrack, avgTrack);
254  total_alhMean += alh_values[0];
255  total_alhMax += alh_values[1];
256  // Beat-cross frequency
257  float bcf_value = K.bcf(aTrack, avgTrack);
258  total_bcf += bcf_value;
259  // Progressive motility
260  String progressMotility_value = "NO";
261  if (str_value > Params.progressMotility) {
262  progressMotility_value = "YES";
263  countProgressiveSperm++;
264  }
265  // DANCE
266  float dance_value = vcl_value * alh_values[0];// vcl*alh_mean
267  total_dance += dance_value;
268  // MAD
269  float mad_value = K.mad(aTrack);
270  total_mad += mad_value;
271 
272  rt.incrementCounter();
273  rt.addValue("Length (frames)", length);
274  rt.addValue("VSL (um/s)", vsl_value);
275  rt.addValue("VCL (um/s)", vcl_value);
276  rt.addValue("VAP (um/s)", vap_value);
277  rt.addValue("LIN", lin_value);
278  rt.addValue("WOB", wob_value);
279  rt.addValue("STR", str_value);
280  rt.addValue("ALH_Mean (um)", alh_values[0]);
281  rt.addValue("ALH_Max (um)", alh_values[1]);
282  rt.addValue("BCF (Hz)", bcf_value);
283  rt.addValue("DANCE (um^2/s)", dance_value);
284  rt.addValue("MAD (degrees)", mad_value);
285  rt.addValue("Progress Motility", progressMotility_value);
286  rt.addValue("Sample", trial.type);
287  rt.addValue("ID", trial.ID);
288  rt.addValue("Source", trial.source);
289  if (!Params.male.isEmpty())
290  rt.addValue("Male", Params.male);
291  if (!Params.date.isEmpty())
292  rt.addValue("Date", Params.date);
293  if (!Params.genericField.isEmpty())
294  rt.addValue("Generic Field", Params.genericField);
295  }
