185  private void calculateMotility(ResultsTable rt, Trial trial) {
186 
187  SignalProcessing sp = new SignalProcessing();
188  Kinematics K = new Kinematics();
189  // Only pass from here tracks with a minimum level of motility
190  List<List<Spermatozoon>> filteredTracks = sp.filterTracksByMotility(trial.tracks);
191  // Calculate values for each track
192  for (ListIterator iT = filteredTracks.listIterator(); iT.hasNext();) {
193  List aTrack = (List) iT.next();
194  List avgTrack = sp.movingAverage(aTrack);
195  float length = (float) aTrack.size();
196  // VSL
197  float vsl_value = K.vsl(aTrack);
198  total_vsl += vsl_value;
199  // VCL
200  float vcl_value = K.vcl(aTrack);
201  total_vcl += vcl_value;
202  // VAP is equivalent to calculate vcl from averaged track
203  float vap_value = K.vcl(avgTrack);
204  total_vap += vap_value;
205  // Linearity
206  float lin_value = (vsl_value / vcl_value) * 100;
207  total_lin += lin_value;
208  // Wobble
209  float wob_value = (vap_value / vcl_value) * 100;
210  total_wob += wob_value;
211  // Straightness
212  float str_value = (vsl_value / vap_value) * 100;
213  total_str += str_value;
214  // Amplitude of lateral head
215  float[] alh_values = K.alh(aTrack, avgTrack);
216  total_alhMean += alh_values[0];
217  total_alhMax += alh_values[1];
218  // Beat-cross frequency
219  float bcf_value = K.bcf(aTrack, avgTrack);
220  total_bcf += bcf_value;
221  // Progressive motility
222  String progressMotility_value = "NO";
223  if (str_value > Params.progressMotility) {
224  progressMotility_value = "YES";
225  countProgressiveSperm++;
226  }
227  // DANCE
228  float dance_value = vcl_value * alh_values[0];// vcl*alh_mean
229  total_dance += dance_value;
230  // MAD
231  float mad_value = K.mad(aTrack);
232  total_mad += mad_value;
233 
234  rt.incrementCounter();
235  rt.addValue("Length (frames)", length);
236  rt.addValue("VSL (um/s)", vsl_value);
237  rt.addValue("VCL (um/s)", vcl_value);
238  rt.addValue("VAP (um/s)", vap_value);
239  rt.addValue("LIN", lin_value);
240  rt.addValue("WOB", wob_value);
241  rt.addValue("STR", str_value);
242  rt.addValue("ALH_Mean (um)", alh_values[0]);
243  rt.addValue("ALH_Max (um)", alh_values[1]);
244  rt.addValue("BCF (Hz)", bcf_value);
245  rt.addValue("DANCE (um^2/s)", dance_value);
246  rt.addValue("MAD (degrees)", mad_value);
