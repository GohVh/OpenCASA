100  void update(ImageProcessor ip) {
101  Random rand = new Random();
102  double epsilon = rand.nextGaussian();
103  // Persistent random walker's differential equation
104  double da = -(beta / ro) * Math.sin(angle) + epsilon * Math.sqrt(2 * Drot);
105  // Update variables
106  angle += da;
107  angle = angle % (2 * Math.PI);
108  float dx = (float) (speed * Math.cos(angle));
109  float dy = (float) (speed * Math.sin(angle));
110  x += dx;
111  y += dy;
112  // Draw Cell
113  ip.fillOval((int) x, (int) y, sizex, sizey);
114 
115  }
