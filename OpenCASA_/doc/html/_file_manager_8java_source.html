134  public List<ImagePlus> loadImageDirectory(String dir) {
135 
136  String[] listOfFiles = getContent(dir);
137  if (listOfFiles == null || listOfFiles.length == 0) {
138  if (listOfFiles != null)
139  JOptionPane.showMessageDialog(null, "Please, select a non-empty folder.");
140  return null;
141  }
142  List<ImagePlus> images = new ArrayList<ImagePlus>();
143  for (int i = 0; i < listOfFiles.length; i++) {
144  IJ.showProgress((double) i / listOfFiles.length);
145  IJ.showStatus("Loading image "+ i +"...");
146  String absoluteFilePath = listOfFiles[i];
147  if(isAVI(absoluteFilePath))
148  continue;
149  String parentsDirectory = getParentDirectory(absoluteFilePath);
150  ImagePlus imp = IJ.openImage(absoluteFilePath);
151  if (imp != null) {
152  imp.setTitle(parentsDirectory + "\\" + imp.getTitle());
153  images.add(imp);
154  }
155  // else - possibly the file is not an image nor AVI
156  }
157  if (images.size() < 1) {
158  JOptionPane.showMessageDialog(null, "Please, select a valid folder.");
159  return null;
160  }
161  return images;
162  }
