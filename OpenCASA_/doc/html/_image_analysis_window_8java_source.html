245  public int selectAnalysis() {
246  // Ask if user wants to analyze a file or directory
247  Object[] options = { "File", "Directory" };
248  String question = "What do you want to analyze?";
249  String title = "Choose one analysis...";
250  final int FILE = 0;
251  final int DIR = 1;
252  final int MULTIDIR = 2;
253  Utils utils = new Utils();
254  int sourceSelection = utils.analysisSelectionDialog(options, question, title);
255  if (sourceSelection < 0) {
256  analysis = TypeOfAnalysis.NONE;
257  return -1;
258  } else if (sourceSelection == FILE) {
259  analysis = TypeOfAnalysis.FILE;
260  } else if (sourceSelection == DIR) {
261  analysis = TypeOfAnalysis.DIRECTORY;
262  }
263  return 0;

275  Dimension screenSize = Toolkit.getDefaultToolkit().getScreenSize();
276  double w = screenSize.getWidth();
277  double h = screenSize.getHeight();
278  int targetWidth = (int) (w * resizeFactor);

280  ImageProcessor ip = impDraw.getProcessor();

282  ip = ip.resize(targetWidth, targetHeight);
283  impDraw.setProcessor(ip);
284  imgLabel.setIcon(new ImageIcon(impDraw.getImage()));
285  imgLabel.repaint();
