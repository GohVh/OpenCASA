110  private void generateResults() {

135  private List<Spermatozoon> getSpermatozoa(Channel rgbChannel) {
136  ComputerVision cv = new ComputerVision();
137 
138  if (rgbChannel == Channel.RED){
139  impTh = cv.getRedChannel(impOrig.duplicate());
140  if(threshold!=-1)
141  threshold = redThreshold;
142  }
143  else if (rgbChannel == Channel.GREEN){
144  impTh = cv.getGreenChannel(impOrig.duplicate());
145  if(threshold!=-1)
146  threshold = greenThreshold;
147  }
148  else if (rgbChannel == Channel.BLUE){
149  impTh = cv.getBlueChannel(impOrig.duplicate());
150  if(threshold!=-1)
151  threshold = blueThreshold;
152  }
153 
154  cv.convertToGrayscale(impTh);
155  thresholdImagePlus(impTh);
156  // this will be useful for painting outlines later
157  if (rgbChannel == Channel.RED)
158  deadImpOutline = impTh;
159  else if (rgbChannel == Channel.GREEN)
160  aliveImpOutline = impTh;
161  else if (rgbChannel == Channel.BLUE)
162  aliveImpOutline = impTh;
163  VideoRecognition vr = new VideoRecognition();
164  List<Spermatozoon>[] sperm = vr.detectSpermatozoa(impTh);
165  return sperm[0];
166  }

196  protected void processImage(boolean eventType) {
197  // If eventType == true, the threshold has changed or it needs to be
198  // calculated
199  // In that class, eventType is always true
200  aliveSpermatozoa = getSpermatozoa(Channel.GREEN);
201  deadSpermatozoa = getSpermatozoa(Channel.RED);
202  if (aliveSpermatozoa != null && deadSpermatozoa != null) {
203  spermatozoa = new ArrayList<Spermatozoon>(aliveSpermatozoa);
204  spermatozoa.addAll(deadSpermatozoa);
205  selectAll();// set as selected all spermatozoa to allow boundary painting
206  idenfitySperm();
207  }
208  // Calculate outlines
209  ComputerVision cv = new ComputerVision();
210  cv.outlineThresholdImage(aliveImpOutline);
211  cv.outlineThresholdImage(deadImpOutline);
212  drawImage();
213  }
