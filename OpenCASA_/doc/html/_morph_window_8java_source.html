173  double total_area = spermatozoon.total_area * Math.pow(Params.micronPerPixel, 2);
174  double total_perimeter = spermatozoon.total_perimeter * Params.micronPerPixel;
175  double total_feret = spermatozoon.total_feret * Params.micronPerPixel;
176  double total_minFeret = spermatozoon.total_minFeret * Params.micronPerPixel;
177  double total_ellipticity = total_feret / total_minFeret;
178  double total_roughness = 4 * Math.PI * total_area / (Math.pow(total_perimeter, 2));
179  double total_elongation = (total_feret - total_minFeret) / (total_feret + total_minFeret);
180  double total_regularity = (Math.PI * total_feret * total_minFeret) / (4 * total_area);

187  morphometrics.addValue("total_perimeter(um)", total_perimeter);
188  morphometrics.addValue("total_length(um)", total_feret);
189  morphometrics.addValue("total_width(um)", total_minFeret);
190  morphometrics.addValue("total_ellipticity", total_ellipticity);
191  morphometrics.addValue("total_roughness", total_roughness);
