145  private void generateResults(Spermatozoon spermatozoon) {
146 
147  ComputerVision cv = new ComputerVision();
148  double total_meanGray = (double) cv.getMeanGrayValue(spermatozoon, impGray, impTh);
149  double total_area = spermatozoon.total_area * Math.pow(Params.micronPerPixel, 2);
150  double total_perimeter = spermatozoon.total_perimeter * Params.micronPerPixel;
151  double total_feret = spermatozoon.total_feret * Params.micronPerPixel;
152  double total_minFeret = spermatozoon.total_minFeret * Params.micronPerPixel;
153  double total_ellipticity = total_feret / total_minFeret;
154  double total_roughness = 4 * Math.PI * total_area / (Math.pow(total_perimeter, 2));
155  double total_elongation = (total_feret - total_minFeret) / (total_feret + total_minFeret);
156  double total_regularity = (Math.PI * total_feret * total_minFeret) / (4 * total_area);
157 
158  morphometrics.incrementCounter();
159  morphometrics.addValue("ID", spermatozoon.id);
160  morphometrics.addValue("Threshold", threshold);
161  morphometrics.addValue("MeanGray", total_meanGray);
162  morphometrics.addValue("Area(um^2)", total_area);
163  morphometrics.addValue("Perimeter(um)", total_perimeter);
164  morphometrics.addValue("Length(um)", total_feret);
165  morphometrics.addValue("Width(um)", total_minFeret);
166  morphometrics.addValue("Ellipticity", total_ellipticity);
167  morphometrics.addValue("Roughness", total_roughness);
168  morphometrics.addValue("Elongation", total_elongation);
169  morphometrics.addValue("Regularity", total_regularity);
170  FileManager fm = new FileManager();
171  morphometrics.addValue("Sample", fm.getParentDirectory(impOrig.getTitle()));
172  morphometrics.addValue("Filename", fm.getFilename(impOrig.getTitle()));
173  if (!Params.male.isEmpty())
174  morphometrics.addValue("Male", Params.male);
175  if (!Params.date.isEmpty())
176  morphometrics.addValue("Date", Params.date);
177  if (!Params.genericField.isEmpty())
178  morphometrics.addValue("Generic Field", Params.genericField);
179  morphometrics.show("Morphometrics");
180  }

194  public boolean isClickInside(Spermatozoon sperm, Point click) {
195  // Get boundaries
196  double offsetX = (double) sperm.bx;
197  double offsetY = (double) sperm.by;
198  int w = (int) sperm.width;
199  int h = (int) sperm.height;
200  // correct offset
201  int pX = (int) (click.getX() - offsetX);
202  int pY = (int) (click.getY() - offsetY);
203  // IJ.log("offsetX: "+offsetX+" ; offsetY: "+offsetY+" ;w: "+w+"; h:
204  // "+h+"px: "+pX+"; py: "+pY);
205  Rectangle r = new Rectangle(w, h);
206  return r.contains(new Point(pX, pY));
207  }

239  public void processImage(boolean eventType) {
240  if (eventType || threshold == -1) {// If true, the threshold has changed or it needs to be calculated
241  ComputerVision cv = new ComputerVision();
242  impTh = impOrig.duplicate();
243  cv.convertToGrayscale(impTh);
244  impGray = impTh.duplicate();
245  thresholdImagePlus(impTh);
246  VideoRecognition vr = new VideoRecognition();
247  List<Spermatozoon>[] sperm = vr.detectSpermatozoa(impTh);
248  if (sperm != null)
249  spermatozoa = sperm[0];
250  // Calculate outlines
251  impOutline = impTh.duplicate();
252  cv.outlineThresholdImage(impOutline);
253  idenfitySperm();
254  }
255  impDraw = impOrig.duplicate();
256  Paint paint = new Paint();
257  paint.drawOutline(impDraw, impOutline);
258  paint.drawBoundaries(impDraw, spermatozoa);
259  setImage();
260  }
