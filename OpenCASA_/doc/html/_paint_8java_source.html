81  public void chemotaxisTemplate(ColorProcessor ip, int numTracks, float chIdx, float slIdx, String sampleID) {
82  // Alpha version of this method
83  ip.setLineWidth(4);
84  // center coords. of the cone used to clasify chemotactic trajectories
85  int xCenter = ip.getWidth() / 2;
86  int yCenter = ip.getHeight() / 2;
87  float upperAngle = (float) (Params.angleDirection + Params.angleAmplitude / 2 + 360) % 360;
88  upperAngle = upperAngle * (float) Math.PI / 180; // calculate and convert
89  // to radians
90  float lowerAngle = (float) (Params.angleDirection - Params.angleAmplitude / 2 + 360) % 360;
91  lowerAngle = lowerAngle * (float) Math.PI / 180; // convert to radians
92  // Upper Line
93  int upperLineX = xCenter + (int) (1000 * Math.cos(upperAngle));
94  int upperLineY = yCenter - (int) (1000 * Math.sin(upperAngle));

97  int lowerLineY = yCenter - (int) (1000 * Math.sin(lowerAngle));

99  ip.moveTo((int) xCenter, (int) yCenter);
100  ip.lineTo((int) upperLineX, (int) upperLineY);

152  public void draw(ImagePlus imp, SerializableList theTracks) {
153 
154  ComputerVision cv = new ComputerVision();
155  cv.convertToRGB(imp);
156  int nFrames = imp.getStackSize();
157  ImageStack stack = imp.getStack();
158  if (imp.getCalibration().scaled()) {
159  IJ.showMessage("MultiTracker", "Cannot display paths if image is spatially calibrated");
160  return;
161  }
162  int upRes = 1;
163  String strPart;
164  int displayTrackNr = 0;
165  SignalProcessing sp = new SignalProcessing();
166  SerializableList avgTracks = sp.averageTracks(theTracks);
167  Kinematics kinematics = new Kinematics();
168  // Draw on each frame
169  for (int iFrame = 1; iFrame <= nFrames; iFrame++) {
170  IJ.showProgress((double) iFrame / nFrames);
171  IJ.showStatus("Drawing Tracks (frame "+iFrame+"/"+nFrames+")...");
172  int yWidth = stack.getWidth();
173  ImageProcessor ip = stack.getProcessor(iFrame);
174  ip.setFont(new Font("SansSerif", Font.PLAIN, 16));
175  displayTrackNr = 0;
176  for (ListIterator iT = theTracks.listIterator(); iT.hasNext();) {
177  List zTrack = (ArrayList) iT.next();
178  displayTrackNr++;
179  ListIterator jT = zTrack.listIterator();
180  Spermatozoon oldSpermatozoon = (Spermatozoon) jT.next();
181  for (; jT.hasNext();) {
182  Spermatozoon newSpermatozoon = (Spermatozoon) jT.next();
183  if (kinematics.getVelocityTrackType(zTrack) == "Slow")
184  ip.setColor(Color.white);
185  else if (kinematics.getVelocityTrackType(zTrack) == "Normal")
186  ip.setColor(Color.yellow);
187  else if (kinematics.getVelocityTrackType(zTrack) == "Fast")
188  ip.setColor(Color.red);
189  // ip.setValue(color);
190  if (Params.drawOrigTrajectories) {
191  ip.moveTo((int) oldSpermatozoon.x * upRes, (int) oldSpermatozoon.y * upRes);
192  ip.lineTo((int) newSpermatozoon.x * upRes, (int) newSpermatozoon.y * upRes);
193  }
194  oldSpermatozoon = newSpermatozoon;
195  // Draw track numbers
196  if (newSpermatozoon.z == iFrame) {
197  strPart = "" + displayTrackNr;
198  ip.setColor(Color.black);
199  // we could do someboundary testing here to place the labels
200  // better when we are close to the edge
201  ip.moveTo((int) (oldSpermatozoon.x / Params.pixelWidth + 0),
202  doOffset((int) (oldSpermatozoon.y / Params.pixelHeight), yWidth, 5));
203  ip.setColor(Color.white);
204  ip.drawString(strPart);
205  }
206  }
207  }
208 // System.out.println("Drawind frame: " + iFrame);
209  }
210  imp.updateAndRepaintWindow();
211  }

256  public void drawChemotaxis(Trial trial, float chIdx, float slIdx) {
257 
258  SignalProcessing sp = new SignalProcessing();
259  SerializableList avgTracks = sp.averageTracks(trial.tracks);
260  // We create another ImageProcesor to draw chemotactic cone and relative
261  // trajectories
262  ColorProcessor ipRelTraj = new ColorProcessor(trial.fieldWidth, trial.fieldHeight);
263  ipRelTraj.setColor(Color.white);
264  ipRelTraj.fill();
265  int xCenter = trial.fieldWidth / 2;
266  int yCenter = trial.fieldHeight / 2;
267  // Draw cone used to clasify chemotactic trajectories
268  ipRelTraj.setColor(Color.green);
269  chemotaxisTemplate(ipRelTraj, avgTracks.size(), chIdx, slIdx, trial.ID);
270  ipRelTraj.setColor(Color.red);
271  ipRelTraj.setLineWidth(4);
272  ipRelTraj.moveTo(xCenter, yCenter);
273  int rx = (int) (1000 * Math.cos(Params.angleDirection * Math.PI / 180));
274  int ry = (int) (1000 * Math.sin(Params.angleDirection * Math.PI / 180));
275  ipRelTraj.lineTo(xCenter + rx, yCenter - ry);
276  ipRelTraj.setLineWidth(1);
277  // Draw average paths
278  IJ.showStatus("Drawing Tracks...");
279  for (ListIterator iT = avgTracks.listIterator(); iT.hasNext();) {
280  List zTrack = (ArrayList) iT.next();
281  ListIterator jT = zTrack.listIterator();
282  Spermatozoon oldSpermatozoon = (Spermatozoon) jT.next();
283  // Variables used to
284  Spermatozoon firstSpermatozoon = new Spermatozoon();
285  firstSpermatozoon.copy(oldSpermatozoon);
286  int xLast = xCenter;
287  int yLast = yCenter;
288  for (; jT.hasNext();) {
289  Spermatozoon newSpermatozoon = (Spermatozoon) jT.next();
290  ipRelTraj.setColor(Color.black);
291  ipRelTraj.moveTo(xLast, yLast);
292  xLast = (int) (xCenter + (newSpermatozoon.x - firstSpermatozoon.x));
293  yLast = (int) (yCenter - (newSpermatozoon.y - firstSpermatozoon.y));
294  ipRelTraj.lineTo(xLast, yLast);
295  oldSpermatozoon = newSpermatozoon;
296  }
297  ipRelTraj.drawOval(xLast - 3, yLast, 6, 6);
298  }
299  new ImagePlus("Chemotactic Ratios", ipRelTraj).show();
300  }

332  public void drawRoseDiagram(int[] histogram, int radius, float chIdx, String sampleID) {
333 
334  // Calculate maximum value of the histogram
335  // to use it later for normalization
336  double max = 0;
337  for (int i = 0; i < histogram.length; i++)
338  if (histogram[i] > max)
339  max = histogram[i];
340  double normFactor = radius / max;
341  int xCenter = radius;
342  int yCenter = radius;
343  ColorProcessor roseDiagram = new ColorProcessor(2 * radius, 2 * radius);
344  roseDiagram.setColor(Color.white);
345  roseDiagram.fill();
346  roseDiagram.setColor(new Color((int) 0, 0, 255, 10));
347  roseDiagram.setLineWidth(1);
348  int NBINS = histogram.length;
349  double angleBin = 2 * Math.PI / (double) NBINS;
350  // Draw on triangle for each bin
351  for (double i = 0; i < NBINS; i++) {
352  int value = histogram[(int) i];
353  int r = (int) (value * normFactor);
354  Polygon p = new Polygon();
355  p.addPoint(xCenter, yCenter); // First vertex
356  int x = (int) (r * Math.cos(i * angleBin));
357  int y = (int) (r * Math.sin(i * angleBin));
358  p.addPoint(xCenter + x, yCenter - y); // Second vertex
359  x = (int) (r * Math.cos((i + 1) * angleBin));
360  y = (int) (r * Math.sin((i + 1) * angleBin));
361  p.addPoint(xCenter + x, yCenter - y); // Third vertex
362  roseDiagram.fillPolygon(p);
363  }
364  roseDiagram.setColor(Color.gray);
365  roseDiagram.setFont(new Font("SansSerif", Font.PLAIN, 22));
366  // Draw line at each 30º
367  for (double i = 0; i < 12; i++) {
368  int x = (int) (radius * Math.cos(i * (2 * Math.PI / 12)));
369  int y = (int) (radius * Math.sin(i * (2 * Math.PI / 12)));
370  roseDiagram.moveTo(xCenter, yCenter);
371  roseDiagram.lineTo(xCenter + x, yCenter - y);
372  roseDiagram.moveTo(xCenter + x, yCenter - y);
373  roseDiagram.drawString("" + i * 30);
374  }
375  roseDiagram.setColor(Color.gray);
376  // Draw three concentric circles as reference values
377  roseDiagram.drawOval(0, 0, 2 * radius, 2 * radius); // First circle
378  roseDiagram.setColor(Color.black);
379  int rx = (int) (radius * Math.cos(Math.PI / 3));
380  int ry = (int) (radius * Math.sin(Math.PI / 3));
381  roseDiagram.moveTo(xCenter + rx, yCenter - ry);
382  roseDiagram.setFont(new Font("SansSerif", Font.PLAIN, 30));
383  roseDiagram.drawString("" + (int) max);// Draw reference value
384  roseDiagram.setColor(Color.gray);
385  roseDiagram.drawOval(radius - 2 * radius / 3, radius - 2 * radius / 3, 4 * radius / 3, 4 * radius / 3); // Second
386  // circle
387  roseDiagram.setColor(Color.black);
388  int r = radius - radius / 3;
389  rx = (int) (r * Math.cos(Math.PI / 3));
390  ry = (int) (r * Math.sin(Math.PI / 3));
391  roseDiagram.moveTo(xCenter + rx, yCenter - ry);
392  roseDiagram.drawString("" + (int) (2 * max / 3));// Draw reference value
393  roseDiagram.setColor(Color.gray);
394  roseDiagram.drawOval(radius - radius / 3, radius - radius / 3, 2 * radius / 3, 2 * radius / 3); // Third
395  // circle
396  roseDiagram.setColor(Color.black);
397  r = radius - 2 * radius / 3;
398  rx = (int) (r * Math.cos(Math.PI / 3));
399  ry = (int) (r * Math.sin(Math.PI / 3));
400  roseDiagram.moveTo(xCenter + rx, yCenter - ry);
401  roseDiagram.drawString("" + (int) (max / 3));// Draw reference value
402  // Draw gradiend direction
403  roseDiagram.setColor(Color.red);
404  roseDiagram.setLineWidth(4);
405  roseDiagram.moveTo(xCenter, yCenter);
406  rx = (int) (radius * Math.cos(Params.angleDirection * Math.PI / 180));
407  ry = (int) (radius * Math.sin(Params.angleDirection * Math.PI / 180));
408  roseDiagram.lineTo(xCenter + rx, yCenter - ry);
409  roseDiagram.setColor(new Color(0, 0, 255, 0));
410  // Draw chemotaxis cone
411  roseDiagram.setColor(Color.green);
412  roseDiagram.setLineWidth(8);
413  double upperAngle = (360 + Params.angleDirection + Params.angleAmplitude / 2) % 360;
414  upperAngle *= Math.PI / 180;
415  int x = (int) (radius * Math.cos(upperAngle));
416  int y = (int) (radius * Math.sin(upperAngle));
417  roseDiagram.moveTo(xCenter, yCenter);
418  roseDiagram.lineTo(xCenter + x, yCenter - y);
419  double lowerAngle = (360 + Params.angleDirection - Params.angleAmplitude / 2) % 360;
420  lowerAngle *= Math.PI / 180;
421  x = (int) (radius * Math.cos(lowerAngle));
422  y = (int) (radius * Math.sin(lowerAngle));
423  roseDiagram.moveTo(xCenter, yCenter);
424  roseDiagram.lineTo(xCenter + x, yCenter - y);
425  // Draw sample info
426  roseDiagram.setLineWidth(1);
427  roseDiagram.setFont(new Font("SansSerif", Font.PLAIN, 30));
428  roseDiagram.moveTo(10, 30);
429  roseDiagram.setColor(Color.blue);
430  roseDiagram.drawString("Sample: " + sampleID);
431  roseDiagram.moveTo(10, 70);
432  roseDiagram.drawString("Ch-Index: " + chIdx);
433 
434  new ImagePlus("Chemotactic Ratios", roseDiagram).show();
435  }
