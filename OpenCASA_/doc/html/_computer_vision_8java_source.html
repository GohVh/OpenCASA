133  float totalGray = 0;
134  float totalPixels = 0;
135  for (int x = bx; x < (width + bx); x++) {
136  IJ.showStatus("scanning pixels...");
137  for (int y = by; y < (height + by); y++) {
138  int pixel = ipTh.get(x, y);
139  if (pixel == 0) {
140  totalGray += (float) ipGray.get(x, y);
141  totalPixels++;
142  }
143  }
144  }
145  return totalGray / totalPixels;

189  public void thresholdImageProcessor(ImageProcessor ip, double lowerThreshold, double upperThreshold) {
190  // Make binary
191  int[] lut = new int[256];
192  for (int j = 0; j < 256; j++) {
193  if (j >= lowerThreshold && j <= upperThreshold)
194  lut[j] = (byte) 0;
195  else
196  lut[j] = (byte) 255;
197  }
198  ip.applyTable(lut);
199  }

206  public void thresholdStack(ImagePlus imp) {
207 
208  ImageStack stack = imp.getStack();
209  ImageProcessor ip = stack.getProcessor(1);
210  ImageStatistics st = ip.getStatistics();
211  double mean = st.mean;
212  double std = st.stdDev;
213  // Set threshold as mean + 2 x standard deviation
214  double lowerThreshold = mean + 2 * std; // std factor: candidate to be a
215  // parameter of the plugin
216  double upperThreshold = 255;
217  // Make binary
218  int[] lut = new int[256];
219  for (int j = 0; j < 256; j++) {
220  if (j >= lowerThreshold && j <= upperThreshold)
221  lut[j] = 0;
222  else
223  lut[j] = (byte) 255;
224  }
225  int nFrames = imp.getStackSize();
226  for (int iFrame = 1; iFrame <= nFrames; iFrame++) {
227  ip = stack.getProcessor(iFrame);
228  ip.applyTable(lut);
