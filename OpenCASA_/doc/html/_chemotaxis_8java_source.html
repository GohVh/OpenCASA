139  private float calculateChIndex(List<List<Spermatozoon>> theTracks) {
140  int trackNr = 0; // Number of tracks
141  int nTracks = theTracks.size();
142  int[] displacements = {0,0};
143  for (List<Spermatozoon> track : theTracks) {
144  IJ.showProgress((double) trackNr / nTracks);
145  IJ.showStatus("Calculating Ch-Index...");
146  trackNr++;
147  int[] instD = countInstantDisplacements(track);
148  displacements[0]+=instD[0];
149  displacements[1]+=instD[1];
150  }
151  float nUpGradient = displacements[0]; // Number of displacements in the gradient direction
152  float nOtherDirs = displacements[1]; // Number of displacements in other direction
153 // IJ.log("nUpGradient: "+nUpGradient+"; nOtherDirs: "+nOtherDirs);
154  float chIdx = 0;
155  if ((nUpGradient + nOtherDirs) > 0) {
156  chIdx = nUpGradient / (nUpGradient + nOtherDirs); // (nUpGradient+nOtherDirs) = Total number of shifts
157  } else {
158  chIdx = -1;
159  }
160  return chIdx; //return index between [0,1]
161  }

169  private float calculateSLIndex(List<List<Spermatozoon>> theTracks) {
170 
171  float nUpGradient = 0; // Number of shifts in the chemoattractant direction
172  float nOtherDirs = 0; // Number of shifts in other direction
173  int trackNr = 0;
174  int nTracks = theTracks.size();
175  double angleChemotaxis = (2 * Math.PI + (Params.angleAmplitude / 2) * Math.PI / 180) % (2 * Math.PI);
176  float ratioSL = 0;
177  for (List<Spermatozoon> aTrack : theTracks) {
178  IJ.showProgress((double) trackNr / nTracks);
179  IJ.showStatus("Calculating SL-Index...");
180  trackNr++;
181  Spermatozoon first = (Spermatozoon) aTrack.get(1);
182  Spermatozoon last = (Spermatozoon) aTrack.get(aTrack.size() - 1);
183  double angle = relativeAngle(first,last); //Between [-PI,PI]
184  //Check if the angle is upGradient or not
185  if (Math.abs(angle) < angleChemotaxis) {
186  nUpGradient++;
187  } else {
188  nOtherDirs++;
189  }
190  }
191  if ((nUpGradient + nOtherDirs) > 0) {
192  ratioSL = nUpGradient / (nUpGradient + nOtherDirs);
193  } else {
194  ratioSL = -1;

244  private int[] countInstantDisplacements(List<Spermatozoon> track) {
245  int nUpGradient = 0;
246  int nOtherDir = 0;
247  int nPoints = track.size();
248  double angleChemotaxis = (2 * Math.PI + (Params.angleAmplitude / 2) * Math.PI / 180) % (2 * Math.PI);
249  for (int j = 0; j < (nPoints - Params.angleDelta); j++) {
250  Spermatozoon oldSpermatozoon = (Spermatozoon) track.get(j);
251  Spermatozoon newSpermatozoon = (Spermatozoon) track.get(j + Params.angleDelta);
252  double angle = relativeAngle(oldSpermatozoon,newSpermatozoon);//Between interval [-PI,PI]
253  if(Params.compareOppositeDirections){//We only take into account angles in the gradient and opposite direction
254  if (Math.abs(angle) < angleChemotaxis) {
255  nUpGradient++;
256  } else if (Math.abs(angle) > (Math.PI - angleChemotaxis)) {
257  nOtherDir++;
258  }
259  }else{//We take into account all angles in all directions
260  if (Math.abs(angle) < angleChemotaxis) {
261  nUpGradient++;
262  } else {
263  nOtherDir++;
264  }
265  }
266  }
267  int[] results = new int[2];
268  results[0] = nUpGradient;
269  results[1] = nOtherDir;
270  return results;
271  }

552  private double relativeAngle(Spermatozoon previous, Spermatozoon next){ //With gradient direction
553  double angleDirection = (2 * Math.PI + Params.angleDirection * Math.PI / 180) % (2 * Math.PI);
554  float diffX = next.x - previous.x;
555  float diffY = next.y - previous.y;
556  double angle = (2 * Math.PI + Math.atan2(diffY, diffX)) % (2 * Math.PI); // Absolute angle
557  angle = (2 * Math.PI + angle - angleDirection) % (2 * Math.PI); // Relative angle between interval [0,2*Pi]
558  if (angle > Math.PI) {
559  angle = -(2 * Math.PI - angle);
560  }
561  return angle; //Between [-PI,PI]
562  }
